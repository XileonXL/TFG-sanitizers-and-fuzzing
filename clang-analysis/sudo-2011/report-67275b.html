882/*
883 * Open sudoers and sanity check mode/owner/type.
884 * Returns a handle to the sudoers file or NULL on error.
885 */
886FILE *
887open_sudoers(const char *sudoers, int doedit, int *keepopen)
888{
889 struct stat statbuf;
890 FILE *fp = NULL((void*)0);
891 int rootstat;
892
893 /*
894 * Fix the mode and group on sudoers file from old default.
895 * Only works if file system is readable/writable by root.
896 */
897 if ((rootstat = stat_sudoerslstat(sudoers, &statbuf)) == 0 &&
898 sudoers_uid == statbuf.st_uid && sudoers_mode != 0400 &&
899 (statbuf.st_mode & 0007777) == 0400) {
900
901 if (chmod(sudoers, sudoers_mode) == 0) {
902 warningx("fixed mode on %s", sudoers);
903 SET(statbuf.st_mode, sudoers_mode)((statbuf.st_mode) |= (sudoers_mode));
904 if (statbuf.st_gid != sudoers_gid) {
905 if (chown(sudoers, (uid_t) -1, sudoers_gid) == 0) {
906 warningx("set group on %s", sudoers);
907 statbuf.st_gid = sudoers_gid;
908 } else
909 warning("unable to set group on %s", sudoers);
910 }
911 } else
912 warning("unable to fix mode on %s", sudoers);
913 }
914
915 /*
916 * Sanity checks on sudoers file. Must be done as sudoers
917 * file owner. We already did a stat as root, so use that
918 * data if we can't stat as sudoers file owner.
919 */
920 set_perms(PERM_SUDOERS0x04);
921
922 if (rootstat != 0 && stat_sudoerslstat(sudoers, &statbuf) != 0)
923 log_error(USE_ERRNO0x02|NO_EXIT0x08, "can't stat %s", sudoers);
924 else if (!S_ISREG(statbuf.st_mode)((((statbuf.st_mode)) & 0170000) == (0100000)))
925 log_error(NO_EXIT0x08, "%s is not a regular file", sudoers);
926 else if ((statbuf.st_mode & 07577) != sudoers_mode)
927 log_error(NO_EXIT0x08, "%s is mode 0%o, should be 0%o", sudoers,
928 (unsigned int) (statbuf.st_mode & 07777),
929 (unsigned int) sudoers_mode);
930 else if (statbuf.st_uid != sudoers_uid)
931 log_error(NO_EXIT0x08, "%s is owned by uid %u, should be %u", sudoers,
932 (unsigned int) statbuf.st_uid, (unsigned int) sudoers_uid);
933 else if (statbuf.st_gid != sudoers_gid)
934 log_error(NO_EXIT0x08, "%s is owned by gid %u, should be %u", sudoers,
935 (unsigned int) statbuf.st_gid, (unsigned int) sudoers_gid);
936 else if ((fp = fopen(sudoers, "r")) == NULL((void*)0))
937 log_error(USE_ERRNO0x02|NO_EXIT0x08, "can't open %s", sudoers);
938 else {
939 /*
940 * Make sure we can actually read sudoers so we can present the
941 * user with a reasonable error message (unlike the lexer).
942 */
943 if (statbuf.st_size != 0 && fgetc(fp) == EOF(-1)) {
944 log_error(USE_ERRNO0x02|NO_EXIT0x08, "can't read %s", sudoers);
945 fclose(fp);
946 fp = NULL((void*)0);
947 }
948 }
949
950 if (fp != NULL((void*)0)) {
951 rewind(fp);
952 (void) fcntl(fileno(fp), F_SETFD2, 1);
953 }
954
955 restore_perms(); /* change back to root */
956 return fp;
957}

1332static char *
1333resolve_editor(char *editor, int nfiles, char **files, char ***argv_out)
1334{
1335 char *cp, **nargv, *editor_path = NULL((void*)0);
1336 int ac, i, nargc, wasblank;
1337
1338 editor = estrdup(editor); /* becomes part of argv_out */
1339
1340 /*
1341 * Split editor into an argument vector; editor is reused (do not free).
1342 * The EDITOR and VISUAL environment variables may contain command
1343 * line args so look for those and alloc space for them too.
1344 */
1345 nargc = 1;
1346 for (wasblank = FALSE0, cp = editor; *cp != '\0'; cp++) {
1347 if (isblank((unsigned char) *cp)((*__ctype_b_loc ())[(int) (((unsigned char) *cp))] & (unsigned
short int) _ISblank))
1348 wasblank = TRUE1;
1349 else if (wasblank) {
1350 wasblank = FALSE0;
1351 nargc++;
1352 }
1353 }
1354 /* If we can't find the editor in the user's PATH, give up. */
1355 cp = strtok(editor, " \t");
1356 if (cp == NULL((void*)0) ||
1357 find_path(cp, &editor_path, NULL((void*)0), getenv("PATH"), 0) != FOUND0) {
1358 efree(editor);
1359 return NULL((void*)0);
1360 }

1362 for (ac = 0; cp != NULL((void*)0) && ac < nargc; ac++) {
1363 nargv[ac] = cp;
1364 cp = strtok(NULL((void*)0), " \t");
1365 }
1366 nargv[ac++] = "--";
1367 for (i = 0; i < nfiles; )
1368 nargv[ac++] = files[i++];
1369 nargv[ac] = NULL((void*)0);

1375/*
1376 * Determine which editor to use. We don't need to worry about restricting
1377 * this to a "safe" editor since it runs with the uid of the invoking user,
1378 * not the runas (privileged) user.
1379 */
1380static char *
1381find_editor(int nfiles, char **files, char ***argv_out)
1382{
1383 char *cp, *editor, *editor_path = NULL((void*)0), **ev, *ev0[4];
1384
1385 /*
1386 * If any of SUDO_EDITOR, VISUAL or EDITOR are set, choose the first one.
1387 */
1388 ev0[0] = "SUDO_EDITOR";
1389 ev0[1] = "VISUAL";
1390 ev0[2] = "EDITOR";
1391 ev0[3] = NULL((void*)0);
1392 for (ev = ev0; *ev != NULL((void*)0); ev++) {
1393 if ((editor = getenv(*ev)) != NULL((void*)0) && *editor != '\0') {
1394 editor_path = resolve_editor(editor, nfiles, files, argv_out);
1395 if (editor_path != NULL((void*)0))
1396 break;
1397 }
1398 }
1399 if (editor_path == NULL((void*)0)) {
1400 /* def_editor could be a path, split it up */
1401 editor = estrdup(def_editor(sudo_defs_table[51].sd_un.str));
1402 cp = strtok(editor, ":");
1403 while (cp != NULL((void*)0) && editor_path == NULL((void*)0)) {
1404 editor_path = resolve_editor(cp, nfiles, files, argv_out);
1405 cp = strtok(NULL((void*)0), ":");
1406 }
1407 if (editor_path)
1408 efree(editor);
1409 }
1410 if (!editor_path) {
1411 audit_failure(NewArgv, "%s: command not found", editor);
1412 warningx("%s: command not found", editor);
1413 }
1414 return editor_path;
1415}
