592 * Generic I/O logging function. Called by the I/O logging entry points.
593 */
594static int
595sudoers_io_log(const char *buf, unsigned int len, int idx)
596{
597 struct timeval now, delay;
598
599 gettimeofday(&now, NULL((void*)0));
600
601 if (sigsetjmp(error_jmp, 1)__sigsetjmp (error_jmp, 1)) {
602 /* called via error(), errorx() or log_error() */
603 return -1;
604 }
605
606#ifdef HAVE_ZLIB_H1
607 if (iolog_compress)
608 gzwrite(io_fds[idx].g, buf, len);
609 else
610#endif
611 fwrite(buf, 1, len, io_fds[idx].f);
612 delay.tv_sec = now.tv_sec;
613 delay.tv_usec = now.tv_usec;
614 timevalsub(&delay, &last_time)do { (&delay)->tv_sec -= (&last_time)->tv_sec; (
&delay)->tv_usec -= (&last_time)->tv_usec; if (
(&delay)->tv_usec < 0) { (&delay)->tv_sec--;
(&delay)->tv_usec += 1000000; } } while (0);
615#ifdef HAVE_ZLIB_H1
616 if (iolog_compress)
617 gzprintf(io_fds[IOFD_TIMING5].g, "%d %f %d\n", idx,
618 delay.tv_sec + ((double)delay.tv_usec / 1000000), len);
619 else
620#endif
621 fprintf(io_fds[IOFD_TIMING5].f, "%d %f %d\n", idx,
622 delay.tv_sec + ((double)delay.tv_usec / 1000000), len);
623 last_time.tv_sec = now.tv_sec;
624 last_time.tv_usec = now.tv_usec;
625
626 return TRUE1;
627}
